303 template<class BaseLayer> class OGRGetNextFeatureThroughRaw
304 {
305 public:
306 
308  OGRFeature* GetNextFeature()
309  {
310  const auto poThis = static_cast<BaseLayer*>(this);
311  while( true )
312  {
313  OGRFeature *poFeature = poThis->GetNextRawFeature();
314  if (poFeature == nullptr)
315  return nullptr;
316 
317  if((poThis->m_poFilterGeom == nullptr
318  || poThis->FilterGeometry( poFeature->GetGeometryRef() ) )
319  && (poThis->m_poAttrQuery == nullptr
320  || poThis->m_poAttrQuery->Evaluate( poFeature )) )
321  {
322  return poFeature;
323  }
324  else
325  delete poFeature;
326  }
327  }
328 };
